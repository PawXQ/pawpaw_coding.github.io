def tesla_calcualte_price_change(ALPHAVANTAGE_URL, ALPHAVANTAGE_PARAMETER):
    tesla_stock_data_responses = requests.get(ALPHAVANTAGE_URL, params=ALPHAVANTAGE_PARAMETER)
    tesla_stock_data_responses.raise_for_status()
    # print(tesla_stock_data_responses.url)
    tesla_price_data = tesla_stock_data_responses.json()
    tesla_price_data_iterator = iter(tesla_price_data["Time Series (Daily)"].items())
    tesla_yesterday_price_data = next(tesla_price_data_iterator)
    tesla_day_before_yesterday_price_data = next(tesla_price_data_iterator)
    # print(tesla_yesterday_price_data)
    # print(tesla_day_before_yesterday_price_data)
    yesterday_close_price = tesla_yesterday_price_data[1]["4. close"]
    day_before_yesterday_close_price = tesla_day_before_yesterday_price_data[1]["4. close"]
    tesla_percentage_range = (float(yesterday_close_price) - float(day_before_yesterday_close_price))/float(day_before_yesterday_close_price) * 100
    tesla_formatted_percentage = f"{tesla_percentage_range:.2f}"
    # tesla_formatted_percentage = -4.95
    return tesla_formatted_percentage


def tesla_news_get(NEWS_API_URL, NEWS_PARAMETER):
    tesla_news_data_responses = requests.get(NEWS_API_URL, params=NEWS_PARAMETER)
    tesla_news_data_responses.raise_for_status()
    tesla_nes_data = tesla_news_data_responses.json()
    # print(tesla_news_data_responses.url)
    return [{"title": article["title"], "url": article["url"], "description": article["description"]} for article in tesla_nes_data["articles"]]
